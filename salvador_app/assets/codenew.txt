import 'package:video_player/video_player.dart';
import 'package:flutter/material.dart';

class HourglassVideo extends StatefulWidget {
  const HourglassVideo({Key? key}) : super(key: key);

  @override
  State<HourglassVideo> createState() => _HourglassVideoState();
}

class _HourglassVideoState extends State<HourglassVideo> {
  late VideoPlayerController _controller;

  @override
  void initState() {
    super.initState();
    _controller = VideoPlayerController.asset("assets/hourglass.mp4")
      ..setLooping(true)
      ..initialize().then((_) {
        setState(() {});
        _controller.play();
      });
  }

  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    if (!_controller.value.isInitialized) {
      return const SizedBox(
        width: 100,
        height: 100,
        child: Center(child: CircularProgressIndicator()),
      );
    }
    return SizedBox(
      width: 100,
      height: 100,
      child: VideoPlayer(_controller),
    );
  }
}
